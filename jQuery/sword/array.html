Array - splice

            var array =['Soldier','Monitor','Platoon','Officer']
            array.splice(2, 1) // ["Platoon"]
            array// ["Soldier", "Monitor", "Officer"]

            array.splice(2,0,'Commander') // []
            array // ["Soldier", "Monitor", "Commander", "Officer"]
